#+begin_src python :results output file :file example.png :output-dir test/

import sys
import numpy as np
import matplotlib.pyplot as plt
from mpl_plotter.two_d import line

x = np.linspace(0, 4*np.pi, 1000)
y = np.sin(5*x)

line(x, y,
     backend="Agg",
     color='purple',
     line_width=1)

plt.savefig(sys.stdout.buffer)
